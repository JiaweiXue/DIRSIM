def compute_k_days_average(ylist, k):
    a = int((k-1)/2)
    b = a+1
    n = len(ylist)
    ylist_new = list()
    for i in range(len(ylist)):
        entries = ylist[max(0, i-a): min(n, i+b)]
        ylist_new.append(np.mean(entries))
    return ylist_new
